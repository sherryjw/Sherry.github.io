vector<int> dailyTemperatures(vector<int>& T) {
    int len = T.size();
    vector<int> result;
    result.resize(len);
    
    stack<int> key;
    stack<int> value;
    
    for (int i = 0; i < len; ++i){
        while (!value.empty() && T[i] > value.top()){
            int t_key = key.top();
            result[t_key] = i-t_key;
            value.pop();
            key.pop();
        }       
        key.push(i);
        value.push(T[i]);
    }
    
    while (!key.empty()){
        int temp = key.top();
        key.pop();
        result[temp] = 0;
    }      
    return result;
}
